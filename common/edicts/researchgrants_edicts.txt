researchgrants_edict_physics_research_10 = {
	length = -1
	resources = {
		category = campaigns
		cost = {
			influence = 0
		}
	}

    prerequisites = {
        "tech_power_plant_1"
        "tech_planetary_unification"
    }

	allow = {
		NOR = {
			has_deficit = energy
            has_deficit = volatile_motes
		}
        has_resource = {
            type = volatile_motes
            amount > 0
        }
	}

	effect = {
        hidden_effect = {
            if = {
                limit = {
                    has_country_flag = researchgrants_no_event_check_flag
                }
                remove_country_flag = researchgrants_no_event_check_flag
            }
        }
		add_modifier = {
            modifier = researchgrants_physics_research_10
        }
	}

	ai_weight = {
		weight = 1
        modifier = {
            factor = 10
            AND = {
                has_resource = {
                    type = energy
                    amount >= 800
                }
                has_monthly_income = {
                    resource = energy
                    value >= 20
                }
                has_monthly_income = {
                    resource = physics_research
                    value >= 16
                }
            }
        }
	}
}
